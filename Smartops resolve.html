<script>

  // ── REPLACE WITH YOUR GOOGLE APPS SCRIPT URL ──

  const APPS_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbwCWEuNLxhx5kr0KxqR_D7ehVXUjCvvcathGQpMD292rXUm9_2HgydDrvskBo2VnKAP/exec";


  function validate() {

    let valid = true;

    const fields = [

      { id: "managerName",  errId: "managerNameErr",  check: v => v.trim().length > 1 },

      { id: "managerEmail", errId: "managerEmailErr", check: v => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v) },

      { id: "taskId",       errId: "taskIdErr",       check: v => v.trim().length > 3 },

      { id: "taskDesc",     errId: "taskDescErr",     check: v => v.trim().length > 3 },

      { id: "taskSite",     errId: "taskSiteErr",     check: v => v !== "" },

      { id: "notes",        errId: "notesErr",        check: v => v.trim().length > 10 },

    ];


    fields.forEach(f => {

      const el = document.getElementById(f.id);

      const err = document.getElementById(f.errId);

      if (!f.check(el.value)) {

        el.classList.add("error");

        err.style.display = "block";

        valid = false;

      } else {

        el.classList.remove("error");

        err.style.display = "none";

      }

    });


    // Check resolution selected

    const resolution = document.querySelector('input[name="resolution"]:checked');

    const resErr = document.getElementById("resolutionErr");

    if (!resolution) {

      resErr.style.display = "block";

      valid = false;

    } else {

      resErr.style.display = "none";

    }


    return valid;

  }


  async function submitResolution() {

    if (!validate()) return;


    const btn = document.getElementById("submitBtn");

    const btnText = document.getElementById("btnText");

    btn.classList.add("loading");

    btnText.textContent = "Submitting...";


    const now = new Date();

    const resolution = document.querySelector('input[name="resolution"]:checked').value;

    const resolvedAt = document.getElementById("resolvedAt").value ||

      now.toISOString().slice(0,16).replace("T"," ");


    const data = {

      type: "resolution",

      managerName: document.getElementById("managerName").value,

      managerEmail: document.getElementById("managerEmail").value,

      taskId: document.getElementById("taskId").value,

      taskDesc: document.getElementById("taskDesc").value,

      taskSite: document.getElementById("taskSite").value,

      resolution,

      notes: document.getElementById("notes").value,

      resolvedAt,

      submittedAt: now.toISOString().slice(0,16).replace("T"," "),

    };


    try {

      await fetch(APPS_SCRIPT_URL, {

        method: "POST",

        mode: "no-cors",

        headers: { "Content-Type": "application/json" },

        body: JSON.stringify(data)

      });

    } catch(e) {}


    // Show summary

    const resLabels = {

      RESOLVED: "✅ Resolved",

      CLOSED: "🚫 Closed",

      REASSIGNED: "🔁 Reassigned",

      ESCALATED: "🚨 Escalated"

    };


    document.getElementById("summaryBox").innerHTML = `

      <div class="summary-row"><span class="summary-label">TASK ID</span><span class="summary-value" style="font-family:var(--mono)">${data.taskId}</span></div>

      <div class="summary-row"><span class="summary-label">RESOLVED BY</span><span class="summary-value">${data.managerName}</span></div>

      <div class="summary-row"><span class="summary-label">ACTION</span><span class="summary-value">${resLabels[resolution]}</span></div>

      <div class="summary-row"><span class="summary-label">SITE</span><span class="summary-value">${data.taskSite}</span></div>

      <div class="summary-row"><span class="summary-label">RESOLVED AT</span><span class="summary-value" style="font-family:var(--mono)">${data.resolvedAt}</span></div>

    `;


    document.getElementById("formBody").style.display = "none";

    document.getElementById("successScreen").style.display = "block";

  }


  function resetForm() {

    document.getElementById("formBody").style.display = "block";

    document.getElementById("successScreen").style.display = "none";

    document.getElementById("submitBtn").classList.remove("loading");

    document.getElementById("btnText").textContent = "Submit Resolution →";

    ["managerName","managerEmail","taskId","taskDesc","taskSite","notes","resolvedAt"].forEach(id => {

      document.getElementById(id).value = "";

      document.getElementById(id).classList.remove("error");

    });

    document.querySelectorAll('input[name="resolution"]').forEach(r => r.checked = false);

  }


  ["managerName","managerEmail","taskId","taskDesc","taskSite","notes"].forEach(id => {

    document.getElementById(id).addEventListener("input", () => {

      document.getElementById(id).classList.remove("error");

    });

  });
